Dockerfile — образ на базе jenkins/inbound-agent:latest
docker-entrypoint.sh — скрипт для запуска DinD

Что установлено в образе:

Docker Engine (CE) + CLI + BuildX + Compose plugin — через официальный apt-репозиторий Docker для Debian
iptables — нужен для сети внутри DinD
Пользователь jenkins добавлен в группу docker, и разрешён sudo dockerd без пароля

Docker-in-Docker:

VOLUME /var/lib/docker — изолированное хранилище для вложенного Docker
docker-entrypoint.sh стартует dockerd в фоне, ждёт его готовности и затем запускает jenkins-agent
Демон слушает на сокете /var/run/docker.sock и на порту 2375

Чтобы запушить образ на DockerHub:
bashdocker build -t your-dockerhub-user/jenkins-slave-dind:latest ./jenkins-slave
docker push your-dockerhub-user/jenkins-slave-dind:latest
Важно при запуске контейнера: нужен флаг --privileged (требование DinD):
bashdocker run --privileged your-dockerhub-user/jenkins-slave-dind:latest

//***************** __Dockerfile-1__ *******************

FROM jenkins/inbound-agent:latest
 
USER root
 
# Install dependencies
RUN apt-get update && apt-get install -y \
    ca-certificates \
    curl \
    gnupg \
    lsb-release \
    iptables \
    sudo \
    && rm -rf /var/lib/apt/lists/*
 
# Install Docker CLI + Engine
RUN install -m 0755 -d /etc/apt/keyrings \
    && curl -fsSL https://download.docker.com/linux/debian/gpg \
       | gpg --dearmor -o /etc/apt/keyrings/docker.gpg \
    && chmod a+r /etc/apt/keyrings/docker.gpg \
    && echo \
       "deb [arch=$(dpkg --print-architecture) signed-by=/etc/apt/keyrings/docker.gpg] \
       https://download.docker.com/linux/debian \
       $(. /etc/os-release && echo "$VERSION_CODENAME") stable" \
       | tee /etc/apt/sources.list.d/docker.list > /dev/null \
    && apt-get update \
    && apt-get install -y \
       docker-ce \
       docker-ce-cli \
       containerd.io \
       docker-buildx-plugin \
       docker-compose-plugin \
    && rm -rf /var/lib/apt/lists/*
 
# Allow jenkins user to run Docker without sudo
RUN usermod -aG docker jenkins \
    && echo "jenkins ALL=(ALL) NOPASSWD: /usr/bin/dockerd" >> /etc/sudoers
 
# Install dind wrapper script
COPY docker-entrypoint.sh /usr/local/bin/docker-entrypoint.sh
RUN chmod +x /usr/local/bin/docker-entrypoint.sh
 
# Docker socket & storage dir for DinD
RUN mkdir -p /var/lib/docker
VOLUME /var/lib/docker
 
USER jenkins
 
ENTRYPOINT ["/usr/local/bin/docker-entrypoint.sh"]

//***************** __Docker entrypoint__ ********************
#!/bin/bash
set -e
 
# Start Docker daemon in the background (DinD)
sudo dockerd \
    --host=unix:///var/run/docker.sock \
    --storage-driver=overlay2 \
    &
 
# Wait for Docker daemon to be ready
echo "Waiting for Docker daemon..."
timeout 30 sh -c 'until docker info > /dev/null 2>&1; do sleep 1; done'
echo "Docker daemon is up."
 
# Hand off to the official Jenkins inbound-agent entrypoint
exec /usr/local/bin/jenkins-agent "$@"
 
//***************** __Commands__ ********************
CREATE IMAGE FOR PART 1
    docker build -t pr2-slave-img .

RUN CONTAINER
    docker run -d --privileged --name pr2-sl1 pr2-slave-img

LOGS
    docker logs pr2-sl1

TEST INTERNAL DOCKER FUNCTIONALITY
    docker exec -it pr2-sl1 docker info

DELETE IMAGE
    docker image rm -f pr2-slave-img:latest

DELETE CONTAINER
    docker container rm -f pr2-sl1

Groovy
pipeline {
    agent {
        label 'infrastructure-slave' // Убедись, что у агента в Jenkins настроена эта метка
    }

    environment {
        // Замени на свои данные DockerHub
        DOCKER_REPO = "твой_логин/private-app-repo"
        DOCKER_CREDS_ID = 'docker-hub-credentials' // ID созданных в Jenkins Credentials
    }

    triggers {
        // Триггер на пуш в DEV (настраивается обычно в самом GitHub Webhook, 
        // но здесь мы ограничиваем выполнение веткой)
        pollSCM('H/2 * * * *') 
    }

    stages {
        stage('Checkout') {
            steps {
                // Проверяем, что мы в ветке DEV
                script {
                    if (env.BRANCH_NAME != 'DEV') {
                        error "Pipeline should only run on DEV branch. Current branch: ${env.BRANCH_NAME}"
                    }
                }
                checkout scm
            }
        }

        stage('Build Image') {
            steps {
                script {
                    // Получаем короткий SHA коммита для тега
                    IMAGE_TAG = sh(script: "git rev-parse --short HEAD", returnStdout: true).trim()
                    echo "Building image with tag: ${IMAGE_TAG}"
                    
                    // Сборка из папки app/
                    sh "docker build -t ${DOCKER_REPO}:${IMAGE_TAG} ./app"
                }
            }
        }

        stage('Push to Registry') {
            steps {
                // Безопасная аутентификация
                withCredentials([usernamePassword(credentialsId: "${DOCKER_CREDS_ID}", 
                                                 usernameVariable: 'DOCKER_USER', 
                                                 passwordVariable: 'DOCKER_PASS')]) {
                    sh "echo ${DOCKER_PASS} | docker login -u ${DOCKER_USER} --password-stdin"
                    
                    echo "Pushing image..."
                    sh "docker push ${DOCKER_REPO}:${IMAGE_TAG}"
                    
                    // Обязательный logout по заданию
                    sh "docker logout"
                }
            }
        }
        
        // Bonus Task 2: PR Creation via Pipeline
        stage('Create PR to MAIN') {
            when {
                expression { return true } // Можно добавить условие успешности предыдущих этапов
            }
            steps {
                // Для этого этапа потребуется GitHub Token, сохраненный в Jenkins
                // и установленный GitHub CLI на агенте (или использование API через curl)
                withCredentials([string(credentialsId: 'github-api-token', variable: 'GITHUB_TOKEN')]) {
                    sh """
                    curl -X POST \
                    -H "Authorization: token ${GITHUB_TOKEN}" \
                    -H "Accept: application/vnd.github.v3+json" \
                    https://api.github.com/repos/${env.GITHUB_USER}/${env.GITHUB_REPO}/pulls \
                    -d '{"title":"Auto-PR from CI: ${IMAGE_TAG}","head":"DEV","base":"MAIN","body":"Automated PR created by Jenkins pipeline after successful CI build."}'
                    """
                }
            }
        }
    }

    post {
        always {
            cleanWs() // Очистка рабочего пространства
        }
        failure {
            echo "Pipeline failed. Check the logs."
        }
    }
}
